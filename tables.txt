drop Table classes;
CREATE TABLE classes(
    SectionID INT NOT NULL,
	QuarterYear VARCHAR(20),
	Meetingtype VARCHAR(5),
	Mandatory int,
	Building VARCHAR(20),
	CourseTitle VARCHAR(20),
	MeetingTime VARCHAR(20),
	EnrollmentLimit INT,
 	Primary Key (SectionID)
);


drop Table courses;
CREATE TABLE Courses(
	CourseId int NOT NULL,
  GradeOption VARCHAR(5) ,
	Lab int ,
	CourseTitle VARCHAR(20),
	Department VARCHAR(20),
	Units INT,
	SectionID INT,
 	flexible INT,
	Primary Key (CourseID)
);

CREATE TABLE Faculty(
	NAME VARCHAR(20) NOT NULL,
	Title VARCHAR(20) NOT NULL,
 	Primary Key (NAME)
);

CREATE TABLE TaughtBy(
	NAME VARCHAR(20) NOT NULL,
	SectionID int NOT NULL,
 	Primary Key (NAME,SectionID)
);

drop Table Schedule;
CREATE TABLE Schedule(
	PID int NOT NULL,
	Sectionid int NOT NULL,
	GRADINGOPTION VARCHAR(20),
  QuarterYear VARCHAR(20) NOT NULL,
	GRADE VARCHAR(20) NOT NULL,
 	Primary Key (PID,SectionID)
);

CREATE TABLE Enrollment(
	PID int NOT NULL,
	CourseId int NOT NULL,
	WAITLIST int NOT NULL,
 	Primary Key (PID,CourseId)
);

CREATE TABLE Student(
	PID int NOT NULL,
  SocialSecurity int NOT NULL,
  MiddleName VARCHAR(20),
  FirstName VARCHAR(20) NOT NULL,
  LastName VARCHAR(20) NOT NULL,
  College VARCHAR(20) NOT NULL,
  Enrolled VARCHAR(20) NOT NULL,
  Major VARCHAR(20) NOT NULL,
  Minor VARCHAR(20),
  Residency VARCHAR(20) NOT NULL,
	Department VARCHAR(20) NOT NULL,
	Program VARCHAR(20) NOT NULL,
	Primary Key (PID)
);

CREATE TABLE ThesisCommittee(
  PID int NOT NULL,
  Candidacy VARCHAR(20) NOT NULL,
  Advisor VARCHAR(20) NOT NULL,
  Primary Key (PID)
);

CREATE TABLE Instruction(
  PID int NOT NULL,
  Name VARCHAR(20) NOT NULL,
  Primary Key (PID)
);

Create Table Probation(
  PID int NOT NULL,
  Reason VARCHAR(20) NOT NULL,
  Primary Key (PID)
);

Create Table ReviewSession(
  SectionID INT NOT NULL,
  Date VARCHAR(20) NOT NULL,
  Time VARCHAR(20) NOT NULL,
  Room VARCHAR(20) NOT NULL,
  Primary Key (SectionID)
);

  Category1 VARCHAR(20),
  UnitsRequired1 INT,
  MinGrade1 VARCHAR(20),
  Category2 VARCHAR(20),
  UnitsRequired2 INT,
  MinGrade2 VARCHAR(20),
  Category3 VARCHAR(20),
  UnitsRequired3 INT,
  MinGrade3 VARCHAR(20),
  Concentration VARCHAR(20),
  ConcentrationUnits INT,
